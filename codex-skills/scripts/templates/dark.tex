% dark.tex — LaTeX preamble for the dark export theme
% Deep navy background, purple accents. Good for screen viewing.
% Used via: pandoc --include-in-header=dark.tex
% NOTE: fontspec, geometry, hyperref are loaded by Pandoc — do NOT reload them here.

% ── Colors ─────────────────────────────────────────────────────────────────
\definecolor{pagebg}{HTML}{1A1A2E}
\definecolor{bodytext}{HTML}{D0D0D0}
\definecolor{headingcolor}{HTML}{A78BFA}
\definecolor{linkcolor}{HTML}{A78BFA}
\definecolor{rulecolor}{HTML}{2D2D5E}
\definecolor{accentpurple}{HTML}{7C3AED}
\definecolor{codebg}{HTML}{16163A}
\definecolor{blockquotebar}{HTML}{7C3AED}
\definecolor{blockquotetext}{HTML}{A0A0C0}

% ── Page Background ───────────────────────────────────────────────────────
\usepackage{pagecolor}
\pagecolor{pagebg}

% ── Base Text ──────────────────────────────────────────────────────────────
\color{bodytext}
\usepackage{setspace}
\setstretch{1.4}
\setlength{\parskip}{0.6em}
\setlength{\parindent}{0pt}

% ── Headings ───────────────────────────────────────────────────────────────
\usepackage{titlesec}

\titleformat{\section}
  {\Large\bfseries\color{headingcolor}}
  {}{0em}{}[\vspace{2pt}{\color{rulecolor}\titlerule[0.4pt]}]
\titlespacing*{\section}{0pt}{1.8em}{0.5em}

\titleformat{\subsection}
  {\large\bfseries\color{headingcolor}}
  {}{0em}{}
\titlespacing*{\subsection}{0pt}{1.5em}{0.4em}

\titleformat{\subsubsection}
  {\normalsize\bfseries\color{headingcolor}}
  {}{0em}{}
\titlespacing*{\subsubsection}{0pt}{1.2em}{0.3em}

% ── Code Blocks ────────────────────────────────────────────────────────────
\usepackage{framed}
\definecolor{shadecolor}{HTML}{16163A}

\makeatletter
\@ifundefined{Shaded}{
  \newenvironment{Shaded}{\begin{snugshade}}{\end{snugshade}}
}{
  \renewenvironment{Shaded}{\begin{snugshade}}{\end{snugshade}}
}
\makeatother

% ── Tables ─────────────────────────────────────────────────────────────────
\renewcommand{\arraystretch}{1.4}

% ── Blockquotes ────────────────────────────────────────────────────────────
\usepackage{mdframed}
\renewenvironment{quote}{%
  \begin{mdframed}[
    topline=false, rightline=false, bottomline=false,
    linewidth=2pt, linecolor=blockquotebar,
    innerleftmargin=10pt, innerrightmargin=0pt,
    innertopmargin=4pt, innerbottommargin=4pt,
    skipabove=\parskip, skipbelow=\parskip,
    backgroundcolor=pagebg
  ]
  \color{blockquotetext}
}{%
  \end{mdframed}
}

% ── Link colors ────────────────────────────────────────────────────────────
\AtBeginDocument{
  \hypersetup{
    linkcolor=linkcolor,
    urlcolor=linkcolor,
    citecolor=linkcolor,
  }
}

% ── Prevent Orphans/Widows ─────────────────────────────────────────────────
\widowpenalty=10000
\clubpenalty=10000

% ── Keep floats close to text ──────────────────────────────────────────────
\usepackage{float}
\floatplacement{table}{H}
\floatplacement{figure}{H}
