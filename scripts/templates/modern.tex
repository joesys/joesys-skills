% modern.tex — LaTeX preamble for the modern export theme
% Muted slate accents, structured layout.
% Used via: pandoc --include-in-header=modern.tex
% NOTE: fontspec, geometry, hyperref are loaded by Pandoc — do NOT reload them here.

% ── Colors ─────────────────────────────────────────────────────────────────
\definecolor{bodytext}{HTML}{374151}
\definecolor{headingcolor}{HTML}{334155}
\definecolor{headingsub}{HTML}{475569}
\definecolor{linkcolor}{HTML}{5B7A9D}
\definecolor{rulecolor}{HTML}{CBD5E1}
\definecolor{accentcolor}{HTML}{94A3B8}
\definecolor{codebg}{HTML}{F8FAFC}
\definecolor{blockquotebar}{HTML}{94A3B8}
\definecolor{blockquotetext}{HTML}{64748B}

% ── Base Text ──────────────────────────────────────────────────────────────
\color{bodytext}
\usepackage{setspace}
\setstretch{1.4}
\setlength{\parskip}{0.6em}
\setlength{\parindent}{0pt}

% ── Headings ───────────────────────────────────────────────────────────────
\usepackage{titlesec}

\titleformat{\section}
  {\Large\bfseries\color{headingcolor}}
  {}{0em}{}[\vspace{2pt}{\color{accentcolor}\titlerule[1pt]}]
\titlespacing*{\section}{0pt}{1.8em}{0.5em}

\titleformat{\subsection}
  {\large\bfseries\color{headingcolor}}
  {}{0em}{}
\titlespacing*{\subsection}{0pt}{1.5em}{0.4em}

\titleformat{\subsubsection}
  {\normalsize\bfseries\color{headingsub}}
  {}{0em}{}
\titlespacing*{\subsubsection}{0pt}{1.2em}{0.3em}

% ── Code Blocks ────────────────────────────────────────────────────────────
\usepackage{framed}
\definecolor{shadecolor}{HTML}{F8FAFC}

\makeatletter
\@ifundefined{Shaded}{
  \newenvironment{Shaded}{\begin{snugshade}}{\end{snugshade}}
}{
  \renewenvironment{Shaded}{\begin{snugshade}}{\end{snugshade}}
}
\makeatother

% ── Tables ─────────────────────────────────────────────────────────────────
\renewcommand{\arraystretch}{1.4}

% ── Blockquotes ────────────────────────────────────────────────────────────
\usepackage{mdframed}
\renewenvironment{quote}{%
  \begin{mdframed}[
    topline=false, rightline=false, bottomline=false,
    linewidth=2pt, linecolor=blockquotebar,
    innerleftmargin=10pt, innerrightmargin=0pt,
    innertopmargin=4pt, innerbottommargin=4pt,
    skipabove=\parskip, skipbelow=\parskip
  ]
  \color{blockquotetext}
}{%
  \end{mdframed}
}

% ── Link colors ────────────────────────────────────────────────────────────
\AtBeginDocument{
  \hypersetup{
    linkcolor=linkcolor,
    urlcolor=linkcolor,
    citecolor=linkcolor,
  }
}

% ── Prevent Orphans/Widows ─────────────────────────────────────────────────
\widowpenalty=10000
\clubpenalty=10000

% ── Keep floats close to text ──────────────────────────────────────────────
\usepackage{float}
\floatplacement{table}{H}
\floatplacement{figure}{H}
